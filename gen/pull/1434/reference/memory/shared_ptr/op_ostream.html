std::operator<<(C++11)

namespace std {
  template <class E, class T, class Y>
  std::basic_ostream<E, T>& operator<<(std::basic_ostream<E, T>& os, const shared_ptr<Y>& p);
}

概要

ストリームに出力する。

効果

ポインタ値を出力する。

os << p.get();

戻り値

os

例外

投げない

例

#include <iostream>
#include <memory>

int main()
{
  std::shared_ptr<int> p(new int(3));

  std::cout << p << std::endl;
}

出力例

0x7f9400d12480
