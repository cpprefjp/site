function
<spanstream>

std::basic_spanbuf::span(C++23)

std::span<charT> span() const noexcept; // (1)

void span(std::span<charT> s) noexcept; // (2)

概要

固定長バッファの取得または設定をする。

効果

  • (1) : std::span 型の固定長バッファを取得する
  • (2) : std::span 型の固定長バッファ s を設定する

例

#include <iostream>
#include <span>
#include <spanstream>

int main()
{
  char buf1[32] = "default";
  std::span<char> span1{buf1};
  std::spanbuf sb{span1};

  // (1)
  std::span<char> read = sb.span();

  std::cout << "(1)" << std::endl;
  std::cout << read.data() << std::endl;
  std::cout << std::endl;

  char buf2[32] = "overwritten";
  std::span<char> span2{buf2};

  // (2)
  sb.span(span2);

  std::cout << "(2)" << std::endl;
  std::cout << sb.span().data() << std::endl;
}

出力

(1)
default

(2)
overwritten
