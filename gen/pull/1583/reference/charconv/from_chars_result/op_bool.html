std::from_chars_result::operator bool(C++26)

constexpr explicit operator bool() const noexcept;

概要

変換が成功したかを判定する。

戻り値

return ec == errc{};

例

#include <iostream>
#include <charconv>

int main()
{
  const char str[] = "123";
  int value{};
  if (std::from_chars(std::begin(str), std::end(str), value)) {
    std::cout << value << std::endl;
  }
  else {
    std::cout << "conversion failed" << std::endl;
  }
}

出力

123
