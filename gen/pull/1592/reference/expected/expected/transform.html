е®ҹйҡӣгҒ«гҒҜиӨҮж•°гӮӘгғјгғҗгғјгғӯгғјгғүгҒҢжҸҗдҫӣгҒ•гӮҢгӮӢгҒҢгҖҒеӨ§гҒҫгҒӢгҒ«гҒҜдёӢиЁҳгӮ·гӮ°гғӢгғҒгғЈгҒ®гӮҲгҒҶгҒ«гҒҝгҒӘгҒӣгӮӢгҖӮ transformгҒёгҒҜгҖҒеј•ж•°гғӘгӮ№гғҲгҒ«1еҖӢгҒ®TеһӢгӮ’гҒЁгӮҠReturnеһӢгӮ’иҝ”гҒҷй–ўж•°гӮ„й–ўж•°гӮӘгғ–гӮёгӮ§гӮҜгғҲгӮ’дёҺгҒҲгӮӢгҖӮ

#include <cassert>
#include <expected>
#include <numeric>
#include <string>
#include <vector>

// 1..Nж•°еҲ—гӮ’з”ҹжҲҗгҒҷгӮӢй–ўж•°
std::vector<int> make_seq(int n)
{
  std::vector<int> seq(n, 0);
  std::iota(seq.begin(), seq.end(), 1);
  return seq;
}

int main()
{
  std::expected<int, std::string> v1 = 3;
  assert((v1.transform(make_seq).value() == std::vector<int>{1,2,3}));

  std::expected<int, std::string> e1 = std::unexpected{"NaN"};
  assert(e1.transform(make_seq).error() == "NaN");
}
