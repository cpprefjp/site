## 例
```cpp example
#include <iostream>
#include <chrono>

namespace chrono = std::chrono;

int main()
{
  const chrono::tzdb_list& ls = chrono::get_tzdb_list();

  int n = std::distance(ls.cbegin(), ls.cend());
  std::cout << n << std::endl;

  for (const chrono::tzdb& tzdb : ls) {
    std::cout << tzdb.version << std::endl;
  }
}
```
