## 例
```cpp example
#include <cassert>
#include <chrono>

using namespace std::chrono;

int main()
{
  const time_zone* tz = zoned_traits<const time_zone*>::locate_zone("Asia/Tokyo");
  assert(tz->name() == "Asia/Tokyo");
}
```
