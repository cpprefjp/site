## 概要

## 例
```cpp example
#include <iostream>
#include <span>
#include <vector>

template<typename I, std::sentinel_for<I> S>
bool is_iter_pair(I, S) {
  return true;
}

bool is_iter_pair(...) {
  return false;
}

int main() {
  std::vector<int> v = {1, 2, 3, 4, 5};
  std::span<int, 5> sp{v};

  auto cit = sp.cend();
  --cit;

  std::cout << *cit << '\n';

  std::cout << std::boolalpha;
  std::cout << is_iter_pair(sp.cbegin(), sp.cend()) << '\n';
  std::cout << is_iter_pair(sp.begin(), sp.cend()) << '\n';
}
```
